1 สร้าง Project ฟรีที่ supabase.com แล้วรัน SQL นี้
CREATE TABLE members ( id bigserial PRIMARY KEY, fid text UNIQUE NOT NULL, nickname text, kid integer, stove_lv integer, note text DEFAULT '', is_mock boolean DEFAULT false, added_at timestamptz DEFAULT now(), updated_at timestamptz DEFAULT now() ); ALTER TABLE members ENABLE ROW LEVEL SECURITY; CREATE POLICY "allow_all" ON members FOR ALL USING (true) WITH CHECK (true);
2 Deploy Edge Function (proxy สำหรับดึงข้อมูลจริง)
# ติดตั้ง Supabase CLI แล้วรันคำสั่งนี้ในโฟลเดอร์โปรเจกต์ supabase functions deploy wos-proxy --no-verify-jwt
📁 ไฟล์ index.ts อยู่ใน supabase/functions/wos-proxy/ (แนบมาพร้อมกัน)
3 ใส่ข้อมูล Project ของคุณ
Edge Function URL = https://[PROJECT_ID].supabase.co/functions/v1/wos-proxy
